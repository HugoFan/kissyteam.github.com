Toolbar For KISSY 

KISSY.use("toolbar,button,menubutton", function (S, Toolbar) {

    var t = new Toolbar({
        render:"#container",
        children:[
            {
                content:"menu-button",
                collapseOnClick:true,
                xclass:'menu-button',
                menu:{
                    xclass:'popupmenu',
                    children:[
                        {
                            content:'item1',
                            xclass:'submenu',
                            menu:{
                                xclass:'popupmenu',
                                children:[
                                    {
                                        content:'item1-1',
                                        xclass:'menuitem'
                                    },
                                    {
                                        content:'item1-2',
                                        xclass:'menuitem'
                                    }
                                ]
                            }
                        },
                        {
                            content:'item2',
                            xclass:'menuitem'
                        }
                    ]
                }
            },
//                {
//                    xclass:'separator'
//                },
            {
                content:'button',
                checkable:true,
                xclass:'button'
            },
            {
                content:"menu-button2",
                xclass:'menu-button',
                collapseOnClick:true,
                menu:{
                    xclass:'popupmenu',
                    children:[
                        {
                            content:'item2-1',
                            xclass:'menuitem'
                        },
                        {
                            content:'item2-2',
                            xclass:'menuitem'
                        }
                    ]
                }
            }
        ]
    }).render();

    t.on("click", function (e) {
        S.log("click : " + e.target.get("content"));
    });

});
