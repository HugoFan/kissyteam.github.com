瀑布示例¶

载入1.2的 kissy

<script src="http://a.tbcdn.cn/s/kissy/1.2.0/kissy.js"></script>

添加css控制

这里的css用来控制瀑布容器中每个瀑布元素的属性

<style>
    .ks-waterfall {
        position:absolute;
        width: 149px ;
        overflow: hidden;
        padding: 15px 15px 15px 4px;
    }
</style>

HTML关键代码

<div>
    <!--放置一个button用于演示销毁瀑布的外部调用-->
    <button id="uninstall">卸载组件</button>
</div>
<div id='container' style="position: relative;">
    <!--这里只需要一个容器就行了，瀑布中每个元素的内容由js配置的模板内容渲染-->
</div>

模板代码

<script type="tpl" id="tpl">
    <div class="ks-waterfall" >
        <img src="{{image_url}}"
             width="{{image_width}}"
             height="{{image_height}}"
        />
        <div class="title">{{title}}</div>
    </div>
</script>

使用脚本生成瀑布

KISSY.use("waterfall", function (S, Waterfall) {
    var waterFall = new Waterfall.Async({
        container: "#container",    //节点容器
        remote: function () {       //函数或者对象，和 ajax 部分一致，只包括 data （服务器端参数）,url（地址），dataType （ajax 类型）
            return {
                data: {
                    from: S.all("#container .ks-waterfall").length
                },
                url: "data.json",
                dataType: "json"
            };
        },
        minColCount: 2,            //最小列数 
        colWidth: 180,             //每列的宽度
        itemTpl: S.all("#tpl").html()  //模板代码，此处从放置代码的dom元素中找到
    });
    
    //外部调用destroy接口卸载瀑布，卸载后将不再有瀑布效果
    S.all("#uninstall").on("click", function () {
        waterFall.destroy();
    });
});
